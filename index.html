Bootstrap Material DatePicker (BS5 - Vanilla JS)

Date Picker

Code

new BootstrapMaterialDatePicker(document.getElementById('date'), { weekStart : 0, time: false });

Time Picker

Code

new BootstrapMaterialDatePicker(document.getElementById('time'), { date: false });

Date Time Picker

Code

new BootstrapMaterialDatePicker(document.getElementById('date-format'), { format : 'dddd DD MMMM YYYY - HH:mm' });

French Locales (Week starts at Monday)

Code

new BootstrapMaterialDatePicker(document.getElementById('date-fr'), { format : 'DD/MM/YYYY HH:mm', lang : 'fr', weekStart : 1, cancelText : 'ANNULER' });

Min Date set

Code

new BootstrapMaterialDatePicker(document.getElementById('min-date'), { format : 'DD/MM/YYYY HH:mm', minDate : new Date() });

Events

Code

var dateEnd = new BootstrapMaterialDatePicker(document.getElementById('date-end'), { weekStart : 0 });
var dateStart = new BootstrapMaterialDatePicker(document.getElementById('date-start'), { weekStart : 0 });
document.getElementById('date-start').addEventListener('change', function(e, date)
{
dateEnd.setMinDate(e.detail.date);
});
